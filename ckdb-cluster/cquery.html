#!/usr/bin/env python3
# -*- coding: utf-8 -*-
import sys
from cassandra.cluster import Cluster

cluster = Cluster(['127.0.0.1'])

session = cluster.connect('contacts')

lastname = sys.argv[1]
query = f"select * from contacts.contacts_lastname_firstname where lastname = '{lastname}'"
rows = session.execute(query)

for user_row in rows:
    print(user_row)
