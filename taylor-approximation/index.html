Taylor Series Sine Approximation

				
					import math, sys
MACHEPS = sys.float_info.epsilon
def mySine(x):
    '''Approximate sin function'''
    # special case: large angle
    if abs(x) > 10**9:
        return float("NaN")
    # special case: small angle
    if x**2 <= MACHEPS:
        return x
    # reduce input argument to |t| <= (pi/2)
    n = round(x/math.pi)
    t = x - n*(math.pi)
    # taylor series calculation
    sol = t - (t**3)/math.factorial(3) + (t**5)/math.factorial(5) - (t**7)/math.factorial(7) + (t**9)/math.factorial(9)\
    - t**11/math.factorial(11) + t**13/math.factorial(13) - t**15/math.factorial(15) + t**17/math.factorial(17)\
    - t**19/math.factorial(19) + t**21/math.factorial(21)

    return sol



def main():
    '''
    demo
    '''
    print(mySine(1.0e-08)) #1e-08
    print(mySine(0.00001)) #9.999999999833334e-06
    print(mySine(0)) #0
    print(mySine(math.pi/2)) #1.0000000000000002
    print(mySine(math.pi)) #-0.0
    print(mySine(100)) #-0.5063656411097555
    print(mySine(-1000)) #-0.8268795405320125
    print(mySine(999999999)) #-0.4101372630100049
    print(mySine(-1000000001)) #nan

if __name__ == "__main__":
    main()
