					def bfs(self, start):
        '''
        enqueues the starting vertex in a queue. While the queue is not empty,
        the algorithm dequeues a vertex from the queue, visits the dequeued vertex,
        enqueues that vertex's adjacent vertices (if not already discovered), and repeats.'''
        if not start in self.map:
            raise ValueError('s')
        frontier_q = deque()
        discovered_q = deque()
        frontier_q.append(start)
        while frontier_q:#while its not empty werid snytax
            current_v = frontier_q.popleft()
            yield current_v#yield the string
            current_v = self.map[current_v]#convert to vertex object
            for adj_v in current_v.adj:
                if not adj_v in discovered_q:
                    frontier_q.append(adj_v)
                    discovered_q.append(adj_v)
