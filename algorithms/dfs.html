Depth-first Search

				
					def dfs(self, start):
        '''
        Return a generator for traversing the graph in depth-first order
		starting from the specified vertex. Raise a ValueError if the vertex does not exist.
		'''
        if not start in self.map:
            raise ValueError('s')
        my_stk = list()
        visited_v = list()
        my_stk.append(start)
        while my_stk:
            curent_v = my_stk.pop()
            if curent_v in visited_v:
                continue
            yield curent_v
            visited_v.append(curent_v)
            curent_v = self.map[curent_v]
            for adj_v in curent_v.adj:
                my_stk.append(adj_v)
