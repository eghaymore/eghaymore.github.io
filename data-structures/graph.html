import math
from collections import deque
class Vertex:
    '''vertex
    '''
    def __init__(self, lbl):
        self.label = lbl#example label: A
        self.adj = {}#adjacent vertices
        self.pred = None#dsp
        self.dist = math.inf#dsp
    def add_adj(self, dest, weight = 0):
        '''
        add adj vertex
        '''
        self.adj[dest] = weight
    def get_wt(self, dest):
        '''get wt'''
        if dest not in self.adj:
            return math.inf
        return self.adj[dest]
class Graph:
    '''
    adj list graph
    '''
    def __init__(self):
        self.size_v = 0
        self.map = {}#contains all vertices

    def add_vertex(self, label):
        '''
        add a vertex with the specified label. Return the graph. label must be a
        string or raise ValueError
        '''
        add_me = Vertex(label)
        if not isinstance(label, str):#check ilegal label
            raise ValueError("no")
        self.map[add_me.label] = add_me#add
        self.size_v += 1
        return self
    def add_edge(self, src, dest, w):
        '''add an edge from vertex src to vertex dest with weight w. Return
        the graph. validate src, dest, and w: raise ValueError if not valid.
        '''
        if not src in self.map or not dest in self.map:#check IMPOrTant: wont work if the vertices are not in the map already.
            raise ValueError("not")
        if not isinstance(w, float)and not isinstance(w, int):#check2
            raise ValueError("does not float")
        one_v = self.map[src]
        one_v.add_adj(dest, w)#add destination vertex to src vertrexs adj list
        return self
    def get_weight(self, src, dest) -> float:
        '''Return the weight on edge src-dest (math.inf if no path exists,
        raise ValueError if src or dest not added to graph).
        '''
        if not src in self.map or not dest in self.map:
            raise ValueError("not")
        its_me = self.map[src]
        return its_me.get_wt(dest)
