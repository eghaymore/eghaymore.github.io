k = sum(merged_df.Oil)
A = np.array([[len(merged_df), k],
              [k, sum(merged_df.Oil**2)]])
b = np.array([[sum(merged_df.Gas)],
              [sum(merged_df.Oil*merged_df.Gas)]])
a_0, a_1 = np.linalg.solve(A,b)
f = lambda x: a_0 + a_1*x
# Overlay f
